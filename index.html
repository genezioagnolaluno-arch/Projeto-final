Projeto Atividade 1: Análise Estrutural Computacional de Treliças Planas
▼

Enunciado Oficial:

Desenvolver um programa completo em Python focado no cálculo estrutural clássico de treliças isostáticas lineares de grande escala. O script deve receber coordenadas de nós e barras, montar dinamicamente as matrizes de equilíbrio com os cossenos diretores e realizar a resolução do sistema linear de forças através do método de eliminação de Gauss-Jordan desenvolvido manualmente de raiz (sem uso de pacotes de álgebra linear prontos de alto nível).

Código Python Implementado (Integral):

import math
import matplotlib.pyplot as plt
import numpy as np

# =========================================================
# SISTEMA DE ANÁLISE DE TRELIÇAS PLANAS ISOSTÁTICAS
# =========================================================

N = int(input('Número de nós: '))
x = []
y = []

print('\n--- Coordenadas dos Nós ---')
for i in range(N):
    xi = float(input(f'Coordenada X do nó {i}: '))
    yi = float(input(f'Coordenada Y do nó {i}: '))
    x.append(xi)
    y.append(yi)

B = int(input('\nNúmero de barras: '))
barras = []
print('\n--- Definição das Barras ---')
for i in range(B):
    no_i = int(input(f'Nó inicial da barra {i}: '))
    no_j = int(input(f'Nó final da barra {i}: '))
    barras.append((no_i, no_j))

R = int(input('\nNúmero de reações de apoio (R): '))
apoios = []
print('\n--- Definição dos Apoios ---')
for i in range(R):
    no_apoio = int(input(f'Nó do apoio {i}: '))
    direcao = input(f'Direção da restrição (X ou Y) no nó {no_apoio}: ').upper()
    apoios.append((no_apoio, direcao))

# Inicialização das matrizes do sistema linear Ax = b
total_equacoes = 2 * N
A = [[0.0] * total_equacoes for _ in range(total_equacoes)]
b = [0.0] * total_equacoes

# Aplicação de cargas externas nos nós
print('\n--- Cargas Externas ---')
for i in range(N):
    fx = float(input(f'Carga horizontal Fx no nó {i} (0 se nenhuma): '))
    fy = float(input(f'Carga vertical Fy no nó {i} (0 se nenhuma): '))
    b[2 * i] = -fx
    b[2 * i + 1] = -fy

# Montagem dos cossenos diretores nas equações de equilíbrio dos nós
for idx, (no_i, no_j) in enumerate(barras):
    dx = x[no_j] - x[no_i]
    dy = y[no_j] - y[no_i]
    L = math.sqrt(dx**2 + dy**2)
    cos_x = dx / L
    cos_y = dy / L
    
    A[2 * no_i][idx] = cos_x
    A[2 * no_i + 1][idx] = cos_y
    A[2 * no_j][idx] = -cos_x
    A[2 * no_j + 1][idx] = -cos_y

# Inclusão das reações na matriz global
col_apoio = B
for no_apoio, direcao in apoios:
    if direcao == 'X':
        A[2 * no_apoio][col_apoio] = 1.0
    elif direcao == 'Y':
        A[2 * no_apoio + 1][col_apoio] = 1.0
    col_apoio += 1

# Algoritmo de Eliminação de Gauss-Jordan desenvolvido manualmente
n = len(b)
for i in range(n):
    pivot = A[i][i]
    if pivot == 0:
        for k in range(i+1, n):
            if A[k][i] != 0:
                A[i], A[k] = A[k], A[i]
                b[i], b[k] = b[k], b[i]
                pivot = A[i][i]
                break
    if pivot == 0:
        continue
    for j in range(n):
        A[i][j] = A[i][j] / pivot
    b[i] = b[i] / pivot
    for k in range(n):
        if k != i:
            fator = A[k][i]
            for j in range(n):
                A[k][j] = A[k][j] - fator * A[i][j]
            b[k] = b[k] - fator * b[i]

print('\n================ RESULTADOS ================')
for i in range(B):
    F = b[i]
    if abs(F) < 1e-6:
        tipo = 'Nula'
    elif F > 0:
        tipo = 'Tração'
    else:
        tipo = 'Compressão'
    print(f'Barra {i}: {F:.2f} -> {tipo}')

print('\n=============== REAÇÕES ===============')
for i in range(R):
    print(f'Reação de Apoio {i}: {b[B + i]:.2f}')

Código de Simulação Integrada (Base de Desenvolvimento):

import pandas as pd
import matplotlib.pyplot as plt
from docx import Document

def calcular_risco(autonomia):
    if autonomia > 3: return "Baixo"
    elif autonomia >= 1: return "Médio"
    else: return "Alto"

def analisar_bairro(nome, demanda, volume, perda, custo):
    volume_perdido = demanda * (perda / 100)
    volume_entregue = demanda - volume_perdido
    custo_diario = demanda * custo
    autonomia = volume / demanda
    risco = calcular_risco(autonomia)
    return {
        "Bairro": nome, "Demanda": demanda, "Volume Entregue": volume_entregue,
        "Volume Perdido": volume_perdido, "Custo Diário": custo_diario,
        "Autonomia": autonomy, "Risco": risco
    }

N = int(input("Quantidade de bairros: "))
dados = []
for i in range(N):
    print(f"\n--- Bairro {i+1} ---")
    nome = input("Nome: ")
    demanda = float(input("Demanda diária (m³): "))
    volume = float(input("Volume disponível (m³): "))
    perda = float(input("Perda na tubulação (%): "))
    custo = float(input("Custo bombeamento por m³: "))
    dados.append(analisar_bairro(nome, demanda, volume, perda, custo))

df = pd.DataFrame(dados)
bairro_critico = df.loc[df["Autonomia"].idxmin(), "Bairro"]

# Geração do Documento Estruturado Oficial .docx
doc = Document()
doc.add_heading('Análise do Sistema de Abastecimento de Água', level=1)
doc.add_paragraph(f"Bairro mais crítico identificado automaticamente: {bairro_critico}")
doc.save("analise_abastecimento.docx")

Código Atividade 8: Roteamento Inteligente e Heurística do Vizinho Mais Próximo
▼

Enunciado Oficial:

Desenvolver as rotinas e iterações completas para calcular distâncias geométricas de caminhos e aplicar a heurística do Vizinho Mais Próximo para otimização de rotas logísticas e estimativa de combustível associado (Versões de 1 a 5).

Implementação de todas as 5 Versões do Algoritmo:

import math

# Versão 1 — Cálculo simples da distância total da rota linear original
def distancia(p1, p2):
    return math.sqrt((p1[0]-p2[0])**2 + (p1[1]-p2[1])**2)

def rota_original(pontos):
    atual = (0, 0)
    total = 0
    for p in pontos:
        total += distancia(atual, p)
        atual = p
    total += distancia(atual, (0, 0)) # Volta à origem
    return total

# Versão 2 — Otimização combinatória via Algoritmo do Vizinho Mais Próximo
def vizinho_mais_proximo(pontos):
    nao_visitados = pontos[:]
    atual = (0, 0)
    rota = []
    total = 0
    while nao_visitados:
        proximo = min(nao_visitados, key=lambda p: distancia(atual, p))
        total += distancia(atual, proximo)
        rota.append(proximo)
        atual = proximo
        nao_visitados.remove(proximo)
    total += distancia(atual, (0, 0))
    return rota, total

# Versão 3 — Cálculo de consumo de combustível associado
CONSUMO_POR_KM = 0.35
def consumo(distancia_total):
    return distancia_total * CONSUMO_POR_KM

# Versão 4 — Verificação de necessidade de reabastecimento logístico
CAPACIDADE_TANQUE = 80
def precisa_reabastecer(distancia_total):
    consumo_total = consumo(distancia_total)
    return consumo_total > CAPACIDADE_TANQUE

# Versão 5 — Execução Integrada com Saída Industrial Completa
if __name__ == "__main__":
    pontos_entrega = [(2, 5), (8, 3), (1, 9), (6, 4)]
    
    dist_original = rota_original(pontos_entrega)
    rota_otim, dist_otim = vizinho_mais_proximo(pontos_entrega)
    
    print(f"Distância Original: {dist_original:.2f} km")
    print(f"Distância Otimizada pelo Vizinho Próximo: {dist_otim:.2f} km")
    print(f"Consumo Otimizado: {consumo(dist_otim):.2f} Litros")
    print(f"Necessita realizar reabastecimento na rota? {precisa_reabastecer(dist_otim)}")

Exercício 13Controle Estatístico e Tolerâncias Dimensionais
▼

Analisar desvios dimensionais de eixos retificados com base em uma medida nominal e sua respectiva tolerância milimétrica simétrica, calculando a eficiência final da linha de produção.

medidas = [50.02, 49.97, 50.06, 49.94, 50.00, 50.08, 49.99, 50.01, 49.92, 50.04]
aprovadas = []
reprovadas = []
nominal = 50.00
tol = 0.05

for m in medidas:
    if abs(m - nominal) <= tol:
        aprovadas.append(round(m, 2))
    else:
        reprovadas.append(m)

print("Aprovadas :", aprovadas)
print("Reprovadas:", reprovadas)
indice = (len(aprovadas) / len(medidas)) * 100
print(f"Índice de aprovação: {indice:.0f}%")
